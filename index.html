@api_app.route('/place-bet', methods=['POST']) def api_place_bet(): req_data = request.get_json() user_id = str(req_data.get('user_id')) items_bet = req_data.get('items') # Mảng các cửa khách chọn, vd: ['bau', 'cua'] amount_per_door = req_data.get('amount', 0) data = load_data() if user_id not in data: return jsonify({'success': False, 'message': 'Người dùng không tồn tại'}) u = data[user_id] total_bet_amount = amount_per_door * len(items_bet) total_bal = u.get('balance', 0) + u.get('locked_balance', 0) if total_bet_amount > total_bal: return jsonify({'success': False, 'message': f'Số dư không đủ! Cần {total_bet_amount:,}đ'}) # 1. Trừ tiền (Ưu tiên ví nạp) temp_bet = total_bet_amount if u['locked_balance'] >= temp_bet: u['locked_balance'] -= temp_bet else: rem = temp_bet - u['locked_balance'] u['locked_balance'] = 0 u['balance'] -= rem # 2. Xúc xắc ngẫu nhiên (hoặc bác có thể thêm logic ép thua tại đây) all_icons = ['bau', 'cua', 'tom', 'ca', 'ga', 'nai'] result = [random.choice(all_icons) for _ in range(3)] # 3. Tính thưởng theo tỷ lệ 1 ăn 1.8 total_win = 0 for bet_item in items_bet: count = result.count(bet_item) if count > 0: # Tỷ lệ 1 ăn 1.8: Thắng 1 con nhận 1.8 * tiền cược cửa đó # Ví dụ đặt 10k trúng 1 con nhận 18k (lãi 8k) total_win += int(amount_per_door * 1.8 * count) if total_win > 0: u['balance'] += total_win log_transaction(u, total_win - total_bet_amount, f"Thắng Bầu Cua Mini App (Đặt {len(items_bet)} cửa)") else: log_transaction(u, -total_bet_amount, f"Thua Bầu Cua Mini App") # 4. Thống kê & Lưu update_stats(u, "bet", total_bet_amount) save_data(data) return jsonify({ 'success': True, 'balance': u['balance'] + u['locked_balance'], 'result': result, 'win': total_win })
